// --- AudioEngine.h ---
struct OSStage {
    const double* coeffs; // 64byteアライメントされた係数へのポインタ
    int numActiveTaps;    // 非ゼロ係数の数
    double* buffer;       // ディレイライン用バッファ（64byteアライメント）
};

// --- AudioEngine.cpp (Audio Thread) ---
void process (const double* input, int numSamples) {
    // 現在の入力fsに基づいて、使用するStageのポインタ配列を選択
    auto* currentStages = getStagesForRate(currentInputFs);
    
    for (int i = 0; i < numActiveStages; ++i) {
        // AVX2/MKL を用いた超高速畳み込み
        applyHalfBandSIMD(input, currentStages[i]);
    }
}
